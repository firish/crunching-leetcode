        strs = sorted(strs, key = lambda k:len(k))
        n = len(strs)
        sw = len(strs[0])
        max_str = ""
        while sw > 0:
            flag = True
            key = strs[0][:sw]
            for i in range(1,n):
                if key not in strs[i]:
                    flag = False
            if flag:
                if len(key)>len(max_str):
                    max_str = key
            sw -= 1
        return max_str
