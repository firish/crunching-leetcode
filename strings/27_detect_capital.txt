We define the usage of capitals in a word to be right when one of the following cases holds:

All letters in this word are capitals, like "USA".
All letters in this word are not capitals, like "leetcode".
Only the first letter in this word is capital, like "Google".
Given a string word, return true if the usage of capitals in it is right.

 

Example 1:

Input: word = "USA"
Output: true
Example 2:

Input: word = "FlaG"
Output: false
 

Constraints:

1 <= word.length <= 100
word consists of lowercase and uppercase English letters.


// my solution
class Solution:
    def detectCapitalUse(self, word: str) -> bool:
        
        if len(word) < 2: return True
        first = 0
        if ord(word[0]) in range(65, 91): first = 1
        second = 0
        if ord(word[1]) in range(65, 91): second = 1
        
        if len(word) == 2:
            if first == 0 and second == 1: return False
            else: return True
            
        for i in range(1, len(word)):
            curr = ord(word[i])
            if first == 1 and second == 1 and curr not in range(65, 91): return False   # rule 1
            if first == 0 and curr in range(65, 91): return False                       # rule 2
            if first == 1 and second == 0 and curr in range(65, 91): return False       # rule 3
        return True
